Main loop tick logic (extends 06.5-04's run_tui):
```rust
loop {
    terminal.draw(|f| draw(f, &app))?;
    if event::poll(Duration::from_millis(500))? {
        if let Event::Key(k) = event::read()? {
            app.handle_key(k.code);
        }
    }
    if app.last_tick.elapsed() >= Duration::from_secs(2) {
        app.refresh();  // re-runs queries + project walk
    }
    if app.should_quit { break; }
}
```
Note: poll timeout dropped from 2000ms to 500ms so keypress latency stays sub-second; the 2s tick is now driven by `last_tick.elapsed()`, not the poll timeout.

       pub fn newest_phase_dir(project_root: &Path) -> Option<(String, u64)> {
           let phases = project_root.join(".planning/phases");
           let entries = std::fs::read_dir(&phases).ok()?;
           let mut best: Option<(String, u64)> = None;
           for e in entries.flatten() {
               if !e.file_type().map(|t| t.is_dir()).unwrap_or(false) { continue; }
               let mtime = e.metadata().ok()?
                   .modified().ok()?
                   .duration_since(std::time::UNIX_EPOCH).ok()?
                   .as_secs();
               let name = e.file_name().to_string_lossy().to_string();
               match &best {
                   Some((_, m)) if *m >= mtime => {}
                   _ => best = Some((name, mtime)),
               }
           }
           best
       }

       pub fn latest_verify_report(vault_path: &Path) -> Option<PathBuf> {
           let dir = vault_path.join("observability/verification-reports");
           let entries = std::fs::read_dir(&dir).ok()?;
           let mut best: Option<(PathBuf, u64)> = None;
           for e in entries.flatten() {
               let path = e.path();
               if path.extension().map(|s| s == "md").unwrap_or(false) {
                   if let Ok(meta) = e.metadata() {
                       if let Ok(modified) = meta.modified() {
                           let ts = modified.duration_since(std::time::UNIX_EPOCH)
                               .map(|d| d.as_secs()).unwrap_or(0);
                           match &best {
                               Some((_, m)) if *m >= ts => {}
                               _ => best = Some((path, ts)),
                           }
                       }
                   }
               }
           }
           best.map(|(p, _)| p)
       }
