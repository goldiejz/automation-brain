pub struct Decision {
    pub ts: String,
    pub class: String,
    pub decision: String,
    pub reason: String,
}

impl Db {
    pub fn recent_decisions(&self, limit: usize) -> Result<Vec<Decision>> {
        let mut stmt = self.conn.prepare(
            "SELECT ts, class, decision, substr(reason,1,80) FROM decisions \
             ORDER BY ts DESC LIMIT ?1"
        )?;
        let rows = stmt.query_map([limit as i64], |r| {
            Ok(Decision {
                ts: r.get(0)?,
                class: r.get(1)?,
                decision: r.get(2)?,
                reason: r.get(3)?,
            })
        })?;
        rows.collect()
    }

    pub fn counts_by_class(&self) -> Result<Vec<(String, i64)>> {
        let mut stmt = self.conn.prepare(
            "SELECT class, COUNT(*) FROM decisions GROUP BY class"
        )?;
        let rows = stmt.query_map([], |r| Ok((r.get(0)?, r.get(1)?)))?;
        rows.collect()
    }
}

// src/sections/portfolio_grid.rs
use ratatui::{Frame, layout::Rect, widgets::{Block, Borders, Row, Table}, style::{Style, Color}};
use crate::sections::Section;
use crate::app::App;

pub struct PortfolioGrid;

impl Section for PortfolioGrid {
    fn name(&self) -> &'static str { "Portfolio" }

    fn render(&self, f: &mut Frame, area: Rect, app: &App) {
        let rows: Vec<Row> = app.projects.iter().map(|p| {
            let health = health_color(p.last_activity_secs);
            Row::new(vec![
                p.name.clone(),
                p.current_phase.clone(),
                relative_time(p.last_activity_secs),
                health_label(p.last_activity_secs),
            ]).style(Style::default().fg(health))
        }).collect();
        let table = Table::new(rows, [
            ratatui::layout::Constraint::Length(30),
            ratatui::layout::Constraint::Length(25),
            ratatui::layout::Constraint::Length(15),
            ratatui::layout::Constraint::Min(10),
        ]).block(Block::default().title("Portfolio").borders(Borders::ALL));
        f.render_widget(table, area);
    }
}

fn health_color(activity_secs: u64) -> Color {
    let now = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .map(|d| d.as_secs()).unwrap_or(0);
    let age = now.saturating_sub(activity_secs);
    if age < 86400 { Color::Green }
    else if age < 86400*7 { Color::Yellow }
    else { Color::Red }
}

pub struct App {
    pub db: Db,
    pub projects: Vec<crate::db::Project>,    // refreshed on tick
    pub recent_decisions: Vec<crate::db::Decision>,
    pub class_counts: Vec<(String, i64)>,
    pub current_section: usize,
    pub should_quit: bool,
    pub last_tick: std::time::Instant,
}

impl App {
    pub fn new(db: Db, portfolio_root: PathBuf) -> Self { ... }
    pub fn refresh(&mut self) {
        // Re-run all queries + project discovery. Called on tick.
        // Errors are swallowed silently → empty Vec (graceful degradation).
        self.projects = crate::db::discover_projects(&self.portfolio_root);
        self.recent_decisions = self.db.recent_decisions(50).unwrap_or_default();
        self.class_counts = self.db.counts_by_class().unwrap_or_default();
        self.last_tick = std::time::Instant::now();
    }
    pub fn handle_key(&mut self, key: KeyCode) {
        match key {
            KeyCode::Char('q') | KeyCode::Esc => self.should_quit = true,
            KeyCode::Char('j') | KeyCode::Down => { /* scroll within section, 06.5-05 adds list state */ }
            KeyCode::Char('k') | KeyCode::Up => { /* scroll up */ }
            KeyCode::Tab => self.current_section = (self.current_section + 1) % 7,
            KeyCode::BackTab => self.current_section = (self.current_section + 6) % 7,
            _ => {}
        }
    }
}

use crossterm::{
    event::{self, DisableMouseCapture, EnableMouseCapture, Event, KeyCode},
    execute,
    terminal::{disable_raw_mode, enable_raw_mode, EnterAlternateScreen, LeaveAlternateScreen},
};
use ratatui::{backend::CrosstermBackend, Terminal};
use std::io;
use std::time::Duration;

fn run_tui(mut app: App) -> io::Result<()> {
    enable_raw_mode()?;
    let mut stdout = io::stdout();
    execute!(stdout, EnterAlternateScreen, EnableMouseCapture)?;
    let backend = CrosstermBackend::new(stdout);
    let mut terminal = Terminal::new(backend)?;

    let result = (|| -> io::Result<()> {
        loop {
            terminal.draw(|f| draw(f, &app))?;
            if event::poll(Duration::from_millis(2000))? {  // 2s tick (D-DASH-REFRESH)
                if let Event::Key(k) = event::read()? {
                    app.handle_key(k.code);
                }
            }
            // 06.5-05 will add: if last_tick.elapsed() >= 2s { app.refresh(); }
            // For this plan, refresh once before the loop is fine.
            if app.should_quit { break; }
        }
        Ok(())
    })();

    // Always restore terminal — even on panic. crossterm::execute! is idempotent.
    disable_raw_mode().ok();
    execute!(io::stdout(), LeaveAlternateScreen, DisableMouseCapture).ok();
    terminal.show_cursor().ok();

    result
}
