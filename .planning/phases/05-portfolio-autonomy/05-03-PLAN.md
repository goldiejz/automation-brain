    2. Define `_portfolio_ceo_load` (private) — reads programme.md once:
       ```bash
       _portfolio_ceo_load() {
         [[ "$_PORTFOLIO_CEO_CACHED" == "1" ]] && return 0
         _PORTFOLIO_CEO_CACHED=1
         local pmd="${ARK_PROGRAMME_MD:-$HOME/vaults/StrategixMSPDocs/programme.md}"
         [[ ! -f "$pmd" ]] && return 0
         # awk: find "## Next Priority" line, then echo the first non-blank,
         # non-comment token on a subsequent line; stop at next "##" heading.
         _PORTFOLIO_CEO_CACHE=$(awk '
           /^##[[:space:]]+Next[[:space:]]+Priority[[:space:]]*$/ { found=1; next }
           found && /^##[[:space:]]/ { exit }
           found {
             # strip leading whitespace + bullet markers
             sub(/^[[:space:]]*[-*][[:space:]]*/, "")
             sub(/^[[:space:]]+/, "")
             # skip blank
             if ($0 == "") next
             # skip pure comments
             if (substr($0, 1, 1) == "#") next
             # take first whitespace-delimited token
             n = split($0, parts, /[[:space:]]+/)
             if (n > 0 && parts[1] != "") {
               # strip trailing punctuation (.,;:)
               gsub(/[.,;:]+$/, "", parts[1])
               print parts[1]
               exit
             }
           }
         ' "$pmd" 2>/dev/null)
         # Lowercase for case-insensitive match (Bash 3 compat: tr, not ${,,})
         _PORTFOLIO_CEO_CACHE=$(echo "$_PORTFOLIO_CEO_CACHE" | tr 'A-Z' 'a-z')
       }
       ```

    3. Define public-ish `_portfolio_ceo_priority <project_path>`:
       ```bash
       _portfolio_ceo_priority() {
         local proj="$1"
         _portfolio_ceo_load

         local base
         base=$(basename "$proj" | tr 'A-Z' 'a-z')
         if [[ "$base" == "$_PORTFOLIO_CEO_CACHE" ]]; then
           echo 1
         else
           echo 0
         fi
       }
       ```
