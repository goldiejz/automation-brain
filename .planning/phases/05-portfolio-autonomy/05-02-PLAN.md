       ```bash
       _portfolio_budget_headroom() {
         local proj="$1"
         local customer
         customer="$(_portfolio_read_customer "$proj")"
         # scratch bucket: no customer cap, treat as 100% headroom
         if [[ "$customer" == "scratch" ]]; then
           echo 100
           return 0
         fi
         local used cap
         used=$(ARK_CUSTOMER="$customer" PROJECT_DIR="" policy_config_get budget.monthly_used 0)
         cap=$(ARK_CUSTOMER="$customer" PROJECT_DIR="" policy_config_get budget.monthly_cap 100000)
         # Defensive: non-numeric → assume cap, headroom 100
         case "$used" in ''|*[!0-9]*) used=0 ;; esac
         case "$cap"  in ''|*[!0-9]*) cap=100000 ;; esac
         if [[ "$cap" -le 0 ]]; then echo 100; return 0; fi
         local pct_used
         pct_used=$(( used * 100 / cap ))
         # ≥80% used → DEFERRED_BUDGET signal (headroom 0)
         if [[ "$pct_used" -ge 80 ]]; then echo 0; return 0; fi
         echo $(( 100 - pct_used ))
       }
       ```
