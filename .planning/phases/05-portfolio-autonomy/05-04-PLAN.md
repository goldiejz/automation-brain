    1. **Cool-down query helper:**
       ```bash
       # Returns 0 (true) if project was DEFERRED_* in last 24h with the same reason class.
       # Args: <project_path> <reason_class>  (reason_class is e.g. "BUDGET" or "HEALTHY")
       _portfolio_recently_deferred() {
         local proj="$1"
         local reason_class="$2"
         local db="${ARK_HOME:-$HOME/vaults/ark}/observability/policy.db"
         [[ ! -f "$db" ]] && return 1   # no history → not cooled
         command -v sqlite3 >/dev/null 2>&1 || return 1
         # 24h ago in ISO 8601 (BSD date first, GNU fallback)
         local cutoff
         cutoff=$(date -u -v-24H +%Y-%m-%dT%H:%M:%SZ 2>/dev/null \
                  || date -u -d '24 hours ago' +%Y-%m-%dT%H:%M:%SZ 2>/dev/null \
                  || echo "")
         [[ -z "$cutoff" ]] && return 1
         local count
         count=$(sqlite3 "$db" \
           "SELECT COUNT(*) FROM decisions
            WHERE class='portfolio'
              AND decision LIKE 'DEFERRED_${reason_class}'
              AND ts >= '$cutoff'
              AND context LIKE '%\"path\":\"$proj\"%';" 2>/dev/null)
         [[ -z "$count" ]] && count=0
         [[ "$count" -gt 0 ]] && return 0
         return 1
       }
       ```

       ```bash
       portfolio_decide() {
         local root="${1:-$ARK_PORTFOLIO_ROOT}"
         # Score everything (no cool-down here — we want full visibility for deferred logging).
         local rows="" path
         while IFS= read -r path; do
           [[ -z "$path" ]] && continue
           local row
           row=$(portfolio_score_project "$path")
           rows+="$row"$'\n'
         done < <(portfolio_scan_candidates "$root")

         # Empty portfolio → NO_CANDIDATE_AVAILABLE
         if [[ -z "${rows//[[:space:]]/}" ]]; then
           _policy_log "portfolio" "NO_CANDIDATE_AVAILABLE" \
             "no projects with .planning/STATE.md under $root" \
             "{\"root\":\"$root\"}"
           return 0
         fi

         # Identify deferred-budget rows: emit DEFERRED_BUDGET audit per row.
         while IFS= read -r row; do
           [[ -z "$row" ]] && continue
           local hr
           hr=$(echo "$row" | awk -F'\t' '{print $6}')
           if [[ "$hr" == "0" ]]; then
             local p ctx
             p=$(echo "$row" | awk -F'\t' '{print $1}')
             ctx=$(_portfolio_row_to_json "$row")
             _policy_log "portfolio" "DEFERRED_BUDGET" \
               "customer over 80% monthly cap" "$ctx"
           fi
         done <<< "$rows"

         # Pick winner (this applies cool-down filtering + sort).
         local winner
         winner=$(portfolio_pick_winner "$root")

         if [[ -z "$winner" ]]; then
           # All filtered → emit DEFERRED_HEALTHY for the top-scoring healthy-but-skipped row.
           # Pick the row with max score that is NOT budget-deferred; if none, NO_CANDIDATE.
           local best_healthy
           best_healthy=$(echo "$rows" | awk -F'\t' '$6 != "0" {print}' \
             | sort -t$'\t' -k8,8nr | head -1)
           if [[ -n "$best_healthy" ]]; then
             local ctx
             ctx=$(_portfolio_row_to_json "$best_healthy")
             _policy_log "portfolio" "DEFERRED_HEALTHY" \
               "no actionable signals (no stuckness, no falling health, no CEO directive)" \
               "$ctx"
           else
             _policy_log "portfolio" "NO_CANDIDATE_AVAILABLE" \
               "all candidates filtered (budget+cooldown)" \
               "{\"root\":\"$root\"}"
           fi
           return 0
         fi

         # SELECTED
         local winning_row ctx
         winning_row=$(echo "$rows" | awk -F'\t' -v p="$winner" '$1 == p {print; exit}')
         ctx=$(_portfolio_row_to_json "$winning_row")
         _policy_log "portfolio" "SELECTED" "highest priority score" "$ctx"
         echo "$winner"
       }
       ```

       ```bash
       _portfolio_row_to_json() {
         local row="$1"
         awk -F'\t' '{
           # Escape backslashes and double quotes in path
           p=$1; gsub(/\\/, "\\\\", p); gsub(/"/, "\\\"", p)
           c=$2; gsub(/\\/, "\\\\", c); gsub(/"/, "\\\"", c)
       ```
