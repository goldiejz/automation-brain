    ```
      learn)
        shift
        if [[ ! -x "$VAULT_PATH/scripts/policy-learner.sh" ]]; then
          echo -e "${RED}❌ scripts/policy-learner.sh missing or not executable${NC}" >&2
          exit 1
        fi
        case "${1:-}" in
          --full)
            bash "$VAULT_PATH/scripts/policy-learner.sh" --full
            ;;
          --since)
            shift
            if [[ -z "${1:-}" ]]; then
              echo -e "${RED}❌ --since requires an ISO8601 date (e.g., 2026-04-19T00:00:00Z)${NC}" >&2
              exit 1
            fi
            bash "$VAULT_PATH/scripts/policy-learner.sh" --since "$1"
            ;;
          "")
            # Default: last 7 days (BSD + GNU date compatible)
            local _since
            _since=$(date -u -v-7d +%Y-%m-%dT%H:%M:%SZ 2>/dev/null \
              || date -u -d '7 days ago' +%Y-%m-%dT%H:%M:%SZ 2>/dev/null \
              || date -u +%Y-%m-%dT%H:%M:%SZ)
            bash "$VAULT_PATH/scripts/policy-learner.sh" --since "$_since"
            ;;
          *)
            echo -e "${RED}❌ Unknown flag: $1${NC}" >&2
            echo "Usage: ark learn [--full | --since ISO8601]" >&2
            exit 1
            ;;
        esac
        ;;
    ```
