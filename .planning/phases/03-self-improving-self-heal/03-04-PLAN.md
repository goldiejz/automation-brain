    ```
    learner_write_digest() {
      local since="${1:-1970-01-01T00:00:00Z}"
      local digest="$VAULT_PATH/observability/policy-evolution.md"
      mkdir -p "$(dirname "$digest")"
      local tmp="${digest}.tmp.$$"

      local now
      now=$(date -u +%Y-%m-%dT%H:%M:%SZ)

      # Score all patterns (re-runs, but cheap)
      local scored
      scored=$(learner_score_patterns "$since")

      {
        echo "# Ark Policy Evolution"
        echo ""
        echo "Generated: $now"
        echo "Window: $since .. $now"
        echo "Audit log: $POLICY_LOG"
        echo ""
        echo "## Promoted patterns (rate ≥ 80%, n ≥ 5)"
        echo ""
        echo "| Pattern | Dispatcher | Complexity | Success rate | Sample size | Decision IDs |"
        echo "|---|---|---|---|---|---|"
        echo "$scored" | jq -r '
          .[]
          | select(.count >= 5 and (.success_count * 100 / .count) >= 80)
          | select(.class != "escalation" and (.class != "budget" or .decision != "ESCALATE_MONTHLY_CAP"))
          | "| \(.class) / \(.decision) | \(.dispatcher) | \(.complexity) | \(.success_count * 100 / .count | floor)% | \(.count) | \(.decision_ids[0:3] | join(", ")) |"
        '
        echo ""
        echo "## Deprecated patterns (rate ≤ 20%, n ≥ 5)"
        echo ""
        echo "| Pattern | Dispatcher | Complexity | Success rate | Sample size | Decision IDs |"
        echo "|---|---|---|---|---|---|"
        echo "$scored" | jq -r '
          .[]
          | select(.count >= 5 and (.success_count * 100 / .count) <= 20)
          | select(.class != "escalation" and (.class != "budget" or .decision != "ESCALATE_MONTHLY_CAP"))
          | "| \(.class) / \(.decision) | \(.dispatcher) | \(.complexity) | \(.success_count * 100 / .count | floor)% | \(.count) | \(.decision_ids[0:3] | join(", ")) |"
        '
        echo ""
        echo "## Mediocre patterns (left alone, 20% < rate < 80%, n ≥ 5)"
        echo ""
        echo "| Pattern | Dispatcher | Complexity | Success rate | Sample size |"
        echo "|---|---|---|---|---|"
        echo "$scored" | jq -r '
          .[]
          | select(.count >= 5)
          | (.success_count * 100 / .count) as $r
          | select($r > 20 and $r < 80)
          | "| \(.class) / \(.decision) | \(.dispatcher) | \(.complexity) | \($r | floor)% | \(.count) |"
        '
        echo ""
        echo "## Recent self_improve audit entries (last 20)"
        echo ""
        if [[ -f "$POLICY_LOG" ]]; then
          grep '"class":"self_improve"' "$POLICY_LOG" 2>/dev/null | tail -20 | jq -r '
            "- `\(.decision_id)` \(.ts) \(.decision) \(.context.class)/\(.context.decision)/\(.context.dispatcher)/\(.context.complexity) (rate \(.context.rate_pct)%, n=\(.context.count))"
          '
        fi
      } > "$tmp"

      mv -f "$tmp" "$digest"
      echo "Digest written: $digest" >&2
    }
    ```
