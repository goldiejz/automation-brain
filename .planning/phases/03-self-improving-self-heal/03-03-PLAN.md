```
_lrn_acquire_lock() {
  local lock="$1"; local timeout="${2:-30}"
  local i=0
  while ! mkdir "$lock" 2>/dev/null; do
    i=$((i+1)); [[ $i -ge $timeout ]] && return 1
    sleep 1
  done
  return 0
}
_lrn_release_lock() { rmdir "$1" 2>/dev/null || true; }
```

    ```
    # === True-blocker re-check (defense in depth) ===
    _lrn_is_true_blocker() {
      local class="$1" decision="$2"
      [[ "$class" == "escalation" ]] && return 0
      [[ "$class" == "budget" && "$decision" == "ESCALATE_MONTHLY_CAP" ]] && return 0
      return 1
    }

    learner_apply_pending() {
      local pending="${1:-$PENDING_FILE}"
      local lock_dir="$VAULT_PATH/.policy-yml.lock"
      [[ ! -s "$pending" ]] && { echo "No pending patches" >&2; return 0; }

      while IFS= read -r line; do
        [[ -z "$line" ]] && continue
        local action class decision dispatcher complexity rate_pct count first_id
        action=$(echo "$line" | jq -r '.action')
        class=$(echo "$line" | jq -r '.class')
        decision=$(echo "$line" | jq -r '.decision')
        dispatcher=$(echo "$line" | jq -r '.dispatcher')
        complexity=$(echo "$line" | jq -r '.complexity')
        rate_pct=$(echo "$line" | jq -r '.rate_pct')
        count=$(echo "$line" | jq -r '.count')
        first_id=$(echo "$line" | jq -r '.decision_ids[0]')

        if _lrn_is_true_blocker "$class" "$decision"; then
          echo "⚠️  Skipping true-blocker: $class/$decision" >&2
          continue
        fi

        if ! _lrn_acquire_lock "$lock_dir" 30; then
          echo "❌ Could not acquire lock at $lock_dir" >&2; return 1
        fi

        # --- Apply patch via python3 ---
        POLICY_YML="$VAULT_PATH/policy.yml" \
        ACTION="$action" CLASS="$class" DECISION="$decision" \
        DISPATCHER="$dispatcher" COMPLEXITY="$complexity" RATE_PCT="$rate_pct" \
        python3 - <<'PY'
import os, sys, re
p = os.environ['POLICY_YML']
action = os.environ['ACTION']
cls = os.environ['CLASS']
dec = os.environ['DECISION']
disp = os.environ['DISPATCHER']
cplx = os.environ['COMPLEXITY']
rate = os.environ['RATE_PCT']

# Try PyYAML, fallback to minimal dotted-key append
try:
    import yaml
    data = {}
    if os.path.exists(p):
        with open(p) as f:
            data = yaml.safe_load(f) or {}
    lp = data.setdefault('learned_patterns', {})
    c = lp.setdefault(cls, {})
    d = c.setdefault(dec, {})
    di = d.setdefault(disp, {})
    cx = di.setdefault(cplx, {})
    if action == 'promote':
        cx['preferred'] = True
        cx.pop('deprecated', None)
    else:
        cx['deprecated'] = True
        cx.pop('preferred', None)
    cx['confidence_pct'] = int(rate)
    with open(p, 'w') as f:
        yaml.safe_dump(data, f, default_flow_style=False, sort_keys=True)
except ImportError:
    # Fallback: append dotted-key lines (lossy but safe for Phase 3 self-test)
    flag = 'preferred' if action == 'promote' else 'deprecated'
    base = f"learned_patterns.{cls}.{dec}.{disp}.{cplx}"
    with open(p, 'a') as f:
        f.write(f"\n# AOS Phase 3 auto-{action}\n")
        f.write(f"{base}.{flag}: true\n")
        f.write(f"{base}.confidence_pct: {rate}\n")
PY

        _lrn_release_lock "$lock_dir"

        # --- Audit log entry via the SINGLE writer ---
        local audit_decision="PROMOTED"
        [[ "$action" == "deprecate" ]] && audit_decision="DEPRECATED"
        local ctx
        ctx=$(printf '{"class":"%s","decision":"%s","dispatcher":"%s","complexity":"%s","rate_pct":%s,"count":%s}' \
          "$class" "$decision" "$dispatcher" "$complexity" "$rate_pct" "$count")
        if type _policy_log >/dev/null 2>&1; then
          _policy_log "self_improve" "$audit_decision" "rate_pct_${rate_pct}_count_${count}" "$ctx" "$first_id" >/dev/null
        else
          echo "⚠️  _policy_log not available; audit entry skipped" >&2
        fi

        # --- Git commit ---
        if git -C "$VAULT_PATH" rev-parse --git-dir >/dev/null 2>&1; then
          git -C "$VAULT_PATH" add policy.yml
          git -C "$VAULT_PATH" commit -m "AOS Phase 3: auto-${action} ${class}/${decision}/${dispatcher}/${complexity} (rate ${rate_pct}%, n=${count})" --quiet || true
        else
          echo "⚠️  Vault is not a git repo; skipping commit" >&2
        fi
      done < "$pending"

      # Archive applied pending file
      mv "$pending" "${pending}.applied-$(date +%s)" 2>/dev/null || true
    }
    ```
